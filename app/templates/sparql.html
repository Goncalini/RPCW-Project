Sample Queries

All Pokemon by Type

PREFIX poke: <http://pokentology.org/ontology/>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

SELECT ?pokemon ?type
WHERE {
  ?pokemon rdf:type poke:Pokemon .
  ?pokemon poke:hasType ?typeEntity .
  ?typeEntity poke:name ?type .
}
ORDER BY ?type ?pokemon

Evolution Chains

PREFIX poke: <http://pokentology.org/ontology/>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

SELECT ?pokemon ?evolvesTo
WHERE {
  ?pokemon poke:evolvesTo ?evolution .
  ?pokemon poke:name ?pokemonName .
  ?evolution poke:name ?evolvesTo .
}
ORDER BY ?pokemon

Pokemon by Ability

PREFIX poke: <http://pokentology.org/ontology/>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

SELECT ?ability (COUNT(?pokemon) as ?count)
WHERE {
  ?pokemon poke:hasAbility ?abilityEntity .
  ?abilityEntity poke:name ?ability .
}
GROUP BY ?ability
ORDER BY DESC(?count)

Type Effectiveness

PREFIX poke: <http://pokentology.org/ontology/>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

SELECT ?attackType ?defendType ?effectiveness
WHERE {
  ?relation rdf:type poke:TypeEffectiveness .
  ?relation poke:attackType ?attackTypeEntity .
  ?relation poke:defendType ?defendTypeEntity .
  ?relation poke:effectivenessMultiplier ?effectiveness .
  ?attackTypeEntity poke:name ?attackType .
  ?defendTypeEntity poke:name ?defendType .
}
ORDER BY ?attackType ?defendType
